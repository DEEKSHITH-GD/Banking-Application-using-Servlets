  `name` varchar(30) NOT NULL,
  `custid` varchar(30) NOT NULL,
  `accno` int NOT NULL,
  `password` varchar(10) CHARACTER SET utf16 COLLATE utf16_bin NOT NULL,
  `balance` int NOT NULL,
  `email` varchar(60) NOT NULL,
  UNIQUE KEY `custid` (`custid`),
  UNIQUE KEY `accno` (`accno`),
  CONSTRAINT `Destination_BankProject_chk_1` CHECK ((`password` <> _utf8mb4''))
